package com.sw.gumi.poly;

public class Parent {
	public int c;
	protected int d;
	private int b;
	public static int X = 10;
	
	
	
	int a = 10;
	void pr() {
		System.out.println("P.a : " + a);
	}
	void m2() {
		System.out.println("m2");
	}
}

class Child extends Parent
{
	int a = 99;
	
	void m1() {
		System.out.println("m1");
	}
	
	@Override
	void pr() {
		System.out.println("C.a : " + a);
	}
}
