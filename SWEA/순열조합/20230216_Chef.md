# SWEA 4012 - 요리사
https://www.acmicpc.net/problem/4386

### 이 문제를 풀기 위한 과정
조합으로 A, B의 요리재료를 나누고, 또 다시한번 각각의 A, B의 재료들의 조합으로 재료의 시너지 합을 구하여야 한다.

CODE1
```java
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.Scanner;
import java.util.StringTokenizer;
import java.io.BufferedReader;
import java.io.FileInputStream;
import java.io.InputStreamReader;

class Solution {
	static int answer;
	static int N;
	static int[][] arr;
	static int[] foodCheck;
	static int[] temp;
	static int sum;
	
	public static void main(String args[]) throws Exception {
		StringBuilder sb = new StringBuilder();
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int T = Integer.parseInt(br.readLine());

		for (int test_case = 1; test_case <= T; test_case++) {
			answer = Integer.MAX_VALUE;
			N = Integer.parseInt(br.readLine());
			foodCheck = new int[N];
			
			arr = new int[N][N];
			for (int i = 0; i < N; i++) {
				StringTokenizer st = new StringTokenizer(br.readLine());
				for (int j = 0; j < N; j++) {
					arr[i][j] = Integer.parseInt(st.nextToken());
				}
			}
			
			for(int i = N - 1; i >= N/2; i--)
			{
				foodCheck[i] = 1;
			}
			
			do {				
				ArrayList<Integer> listA = new ArrayList<>();
				ArrayList<Integer> listB = new ArrayList<>();
				
				for(int i = 0; i < foodCheck.length; i++)
				{
					if(foodCheck[i] == 0)
					{
						listA.add(i);
					} else
					{
						listB.add(i);
					}
				}
				
				sum = 0;
				temp = new int[2];
				synergy(0, 0, listA);
				int synergyA = sum;
				
				sum = 0;
				temp = new int[2];
				synergy(0, 0, listB);
				int synergyB = sum;
				
				answer = Math.min(Math.abs(synergyA - synergyB), answer);
			
			}while(nextPermutation(foodCheck));
			
			
			
			sb.append("#").append(test_case).append(" ").append(answer).append("\n");
			
		}
		System.out.println(sb.toString());
	}
	
	static void synergy(int depth, int idx, ArrayList<Integer> list)
	{
		if(depth == 2)
		{
			int x = temp[0];
			int y = temp[1];
			
			sum += arr[x][y];
			sum += arr[y][x];
			return;
		}
		
		for(int i = idx; i < N/2; i++)
		{
			temp[depth] = list.get(i);
			synergy(depth + 1, i + 1, list);
		}
	}
	
	static boolean nextPermutation(int[] input)
	{
		int len = input.length;
		
		int top = len - 1;
		
		while(top > 0 && input[top - 1] >= input[top] )
			top--;
		
		if(top == 0) return false;
		
		// step2. 꼭대기 바로 앞(i-1) 자리에 교환할 값을 뒤쪽부터 찾는다.
		int target = len - 1;
		while(input[top - 1] >= input[target]) 
			target--;
		
		// step3. 꼭대기 바로 앞(i-1)자리와 그 자리 값보다 한 단계 큰 자리(j) 수와 교환
		swap(input, top - 1, target);
		
		// step4. 꼭대기부터 맨 뒤까지 오름차순으로 정렬
		int last = len - 1;
		while(top < last)
			swap(input, top++, last--);
		
		return true;
	}
	
	static void swap(int[] input, int i, int j)
	{
		int temp = input[i];
		input[i] = input[j];
		input[j] = temp;
	}
	
}
```
