#  백준 - 가장 긴 증가하는 부분 수열2

### 이 문제를 풀기 위한 과정


CODE1

    #include <iostream>
    #include <algorithm>
    #include <vector>

    using namespace std;

    int solution(int N, vector<int> v);

    int main(void)
    {
        int N;
        cin >> N;

        vector<int> v;
        for (int i = 0; i < N; i++)
        {
            int num;
            cin >> num;
            v.push_back(num);
        }

        int answer = solution(N, v);

        cout << answer;
    }

    int solution(int N, vector<int> v)
    {
        int answer = -1;

        vector<int> vec;
        vec.push_back(0);

        for (int i = 0; i < N; i++)
        {
            if (v[i] > vec.back())
            {
                vec.push_back(v[i]);
            }
            else
            {
                int idx = lower_bound(vec.begin(), vec.end(), v[i]) - vec.begin();
                vec[idx] = v[i];
            }
        }

        answer = vec.size() - 1;

        return answer;
    }
